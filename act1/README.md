######
 Install bash and vim:
   command:
     - apk add bash
     - apk add vim

 Install git:
   command:
     - apk add git

 Create a directory name act1:
   command:
     - mkdir act1

 Initialize git
   command:
     - git init

 To push clone the git
   command:
     - git clone "https://www.github.com/username/repository.git"

 To push the files to git
   command:
     - git add .
     - git commit -m "name"
     - git push -u origin "branch"
